```typescript
import { query } from './db';
import type { Layout } from 'react-grid-layout';

export interface WorkspaceLayout {
  schemaVersion: number;
  items: Layout[];
  pluginConfigs: Record<string, Record<string, unknown>>;
}

// D-10: Daily Ops preset — default first-visit layout
export const DAILY_OPS_PRESET: WorkspaceLayout = {
  schemaVersion: 1,
  items: [
    { i: 'optimus.today-brief',    x: 0, y: 0, w: 6, h: 8 },
    { i: 'optimus.approval-queue', x: 6, y: 0, w: 6, h: 8 },
    { i: 'optimus.agent-status',   x: 0, y: 8, w: 12, h: 6 },
  ],
  pluginConfigs: {},
};

// D-15: Migration function — runs before any layout is used
export function migrateWorkspace(raw: unknown): WorkspaceLayout {
  const data = raw as Partial<WorkspaceLayout>;
  const version = data?.schemaVersion ?? 0;

  // Phase 1: v1 is baseline. Future phases add migration steps here.
  // Example: if (version < 2) { /* migrate v1 -> v2 */ }

  return {
    schemaVersion: 1,
    items: Array.isArray(data?.items) ? data.items : DAILY_OPS_PRESET.items,
    pluginConfigs: (data?.pluginConfigs && typeof data.pluginConfigs === 'object') 
      ? data.pluginConfigs 
      : {},
  };
}

// Get workspace for a member. Falls back to Daily Ops preset.
export async function getWorkspace(
  memberId: string,
  name: string = 'Daily Ops'
): Promise<WorkspaceLayout> {
  const result = await query(
    'SELECT layout, schema_version FROM board.workspaces WHERE member_id = $1 AND name = $2',
    [memberId, name]
  );

  if (result.rows.length === 0) {
    return DAILY_OPS_PRESET;
  }

  const row = result.rows[0];
  return migrateWorkspace(row.layout);
}

// Save workspace layout (upsert). Per D-14: called by debounced auto-save.
export async function saveWorkspace(
  memberId: string,
  name: string,
  layout: WorkspaceLayout
): Promise<void> {
  await query(
    `INSERT INTO board.workspaces (member_id, name, layout, schema_version, updated_at)
     VALUES ($1, $2, $3, $4, now())
     ON CONFLICT (member_id, name)
     DO UPDATE SET layout = $3, schema_version = $4, updated_at = now()`,
    [memberId, name, JSON.stringify(layout), layout.schemaVersion]
  );
}
```

All queries use parameterized `$1, $2` — no string interpolation (per P4, CLAUDE.md).

**Step 2: Create board/src/app/api/workspaces/route.ts:**

Follow the existing API route pattern from `board/src/app/api/ops/route.ts`. Auth check using the board's existing session retrieval method (read `auth.ts` and `next-auth-options.ts` to find the exact function — likely `getServerSession` with `authOptions`).

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { getServerSession } from 'next-auth';
import { authOptions } from '@/lib/next-auth-options';
import { getWorkspace, saveWorkspace, migrateWorkspace } from '@/lib/workspaces';
import type { WorkspaceLayout } from '@/lib/workspaces';

export async function GET(req: NextRequest) {
  const session = await getServerSession(authOptions);
  if (!session?.user?.name) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
  }

  const name = req.nextUrl.searchParams.get('name') ?? 'Daily Ops';
  const workspace = await getWorkspace(session.user.name, name);
  return NextResponse.json(workspace);
}

export async function PUT(req: NextRequest) {
  const session = await getServerSession(authOptions);
  if (!session?.user?.name) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
  }

  const body = await req.json();
  const layout = migrateWorkspace(body.layout);
  const name = (body.name as string) ?? 'Daily Ops';

  await saveWorkspace(session.user.name, name, layout);
  return NextResponse.json({ ok: true });
}
```

Per P1 (deny by default): session check returns 401 before any DB access.
Per P2 (infrastructure enforces): auth is in the API route, not in the client component.
`member_id` is always `session.user.name` — never trust client-supplied member_id.

IMPORTANT: Read `board/src/lib/auth.ts` and `board/src/lib/next-auth-options.ts` first to determine the exact session retrieval pattern used in this codebase. The example above uses `getServerSession(authOptions)` which is the standard next-auth v4 pattern — adjust if the codebase uses a wrapper.

**Add a custom useDebounce hook** (inline in GridArea.tsx or in a separate utility — keep it simple per research recommendation):

```typescript
function useDebounce<T>(value: T, delay: number): T {
  const [debounced, setDebounced] = useState(value);
  useEffect(() => {
    const timer = setTimeout(() => setDebounced(value), delay);
    return () => clearTimeout(timer);
  }, [value, delay]);
  return debounced;
}
```

**Load workspace on mount:**
- On component mount (inside a useEffect), `fetch('/api/workspaces?name=Daily+Ops')` to get the saved layout
- If the fetch succeeds and returns layout data, call `setLayout(response.items)` to replace the default layout
- If the fetch fails (network error, 401, no saved workspace), silently fall back to DAILY_OPS_DEFAULT_LAYOUT already in state
- Per UI-SPEC Copywriting Contract: no user-visible error on workspace load failure — "Could not load your workspace. Loading default layout." is planned but silent recovery to preset is the Phase 1 behavior
